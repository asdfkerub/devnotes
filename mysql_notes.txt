SELECT
  SELECT * FROM table_name    #select everything from a table
  SELECT column_name FROM table_name    #selects a column from a table
  SELECT column_1 , column_2 FROM table_name    #select column_1 and column_2 from table
  SELECT column_1 FROM table_name WHERE id = 3    #select column_1 from table of a row
  SELECT column_1 FROM table_name WHERE id = 3 or id = 2    #select column_1 from table of row 3 or row 2
  SELECT column_2 FROM table_name WHERE id > 3    #select column_1 from table of rows greater than 3
  SELECT * FROM table_name WHERE column_1 LIKE '%k'     #select everything from table where column_1 values ends with a ' K '
  SELECT * FROM table_name WHERE column_1 LIKE 'K%'     #select everything from table where column values starts with a ' K '
