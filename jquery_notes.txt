return false;
 - makes the default to return false. For example: if you submit a form but
   doesnt want to go to a page on submit.

.submit();
 - you can use submit to hard submit a form with a click somewhere outside of the <form>
 - you can use submit to do something when it is submitted like .submit(function(){//code here});

.serialize()
 - preps a form to be handled by a computer. preps to the from to be manipulated
   behind the scenes.
 - takes information inside the submitted form to be in one string and separated by end signs
   use console.log($('form name').serialize()); to see what serialize do.

.children()
 - targets the childrens inside of the containing element
    ForExample: $('div').children('p').css('color','red');

.parent()
 - targets the parent or the container element
    ForExample: $('p').parent().append('<p>new</p>');

.siblings()
 - targets the siblings.
   ForExample:
     $('p').click(function(){
       $('this').siblings('h3').css('color','blue');
     });

event.stopPropagation();
 - doesnt allow function to take affect to the parent(container) element.
   For Example:
    $().click(function(anyName){
      anyName.stopPropagation();
    })
